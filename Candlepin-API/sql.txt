consumer_params:
---------------------------------------------------------------------------------------------------------------------------------
select f.element as fact, cp.uuid, t.label, o.account, cp.username from cp_consumer cp , cp_consumer_facts f , cp_owner o, cp_consumer_type t where f.cp_consumer_id = cp.id and f.mapkey = 'virt.uuid'  and o.id = cp.owner_id and cp.type_id = t.id limit 10;
---------------------------------------------------------------------------------------------------------------------------------
Virt_Guest.csv
---------------------------------------------------------------------------------------------------------------------------------
select distinct cp1.uuid from cp_consumer_guests cg, cp_consumer cp1 , cp_consumer_facts cf1 , cp_consumer cp2 , cp_consumer_facts cf2  where cp1.id = cg.consumer_id and cg.guest_id = cf1.element and cf1.mapkey = 'virt.uuid' and cp1.uuid <> cp2.uuid and cp2.id = cf2.cp_consumer_id and cf2.mapkey = 'virt.uuid'  limit 10;
---------------------------------------------------------------------------------------------------------------------------------
consumers_with_ents.csv 
---------------------------------------------------------------------------------------------------------------------------------
select b.uuid, a.id  from cp_entitlement a, cp_consumer b where b.id = a.consumer_id  limit 10;
---------------------------------------------------------------------------------------------------------------------------------
consumers_with_available_pools.csv
---------------------------------------------------------------------------------------------------------------------------------
select cp.uuid ,e.pool_id  from cp_entitlement e,cp_consumer cp where cp.id = e.consumer_id limit 10;
---------------------------------------------------------------------------------------------------------------------------------
Virt_Host.csv 
---------------------------------------------------------------------------------------------------------------------------------
select distinct cp.uuid from cp_consumer_facts cf , cp_consumer cp where cp.id = cf.cp_consumer_id and  cf.element in (select guest_id from cp_consumer_guests  where guest_id <> '') limit 10;
